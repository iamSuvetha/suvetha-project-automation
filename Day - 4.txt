2. Write a method which should join the given string and number


public class StringNumberJoin {
    
    public static String joinStringAndNumber(String text, int number) {
        return text + number;    
}

    public static void main(String[] args) {
        String result = joinStringAndNumber("Hi", 5);
        System.out.println(result);
    }
}
